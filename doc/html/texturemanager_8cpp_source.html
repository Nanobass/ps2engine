87void GetOpenGLFormatAndType(GS::tPSM psm, GLenum& format, GLenum& type)
88{
89 switch(psm)
90 {
91 case GS::kPsm32:
92 format = GL_RGBA;
93 type = GL_UNSIGNED_BYTE;
94 break;
95 case GS::kPsm24:
96 format = GL_RGB;
97 type = GL_UNSIGNED_BYTE;
98 break;
99 case GS::kPsm16:
100 format = GL_RGBA;
101 type = GL_UNSIGNED_SHORT_5_5_5_1;
102 break;
103 case GS::kPsm8:
104 format = GL_COLOR_INDEX;
105 type = GL_UNSIGNED_BYTE;
106 break;
107 default:
108 break;
109 }
110}

114texture_ptr texture_manager::load_gs_texture(const memory::resource_id& id, const std::string& path)
115{
116 std::ifstream resource(path, std::ios_base::in);
117 GsTextureHeader header;
118 resource.read((char*) &header, sizeof(GsTextureHeader));
119
120 assert(header.mMagic == GsTextureHeader::MAGIC);
121
122 GLenum format = 0, type = 0;
123 PS2::GetOpenGLFormatAndType((GS::tPSM) header.mPsm, format, type);
124 uint8_t bpp = 0;
125 if(header.mPsm == GS_PSM_4) bpp = 4;
126 if(header.mPsm == GS_PSM_8) bpp = 8;
127 if(header.mPsm == GS_PSM_16) bpp = 16;
128 if(header.mPsm == GS_PSM_24) bpp = 24;
129 if(header.mPsm == GS_PSM_32) bpp = 32;
130
131 texture_buffer core(header.mWidth, header.mHeight, bpp, format, type);
132 resource.read((char*) core.mData.data(), core.mData.size());
133
134 if( header.mPsm == GS_PSM_8 || header.mPsm == GS_PSM_4 )
135 {
136 PS2::GetOpenGLFormatAndType((GS::tPSM) header.mPsm, format, type);
137 if(header.mClutPsm == GS_PSM_16) bpp = 16;
138 if(header.mClutPsm == GS_PSM_24) bpp = 24;
139 if(header.mClutPsm == GS_PSM_32) bpp = 32;
140 texture_buffer clut(header.mClutWidth, header.mClutHeight, bpp, format, type);
141 resource.read((char*) clut.mData.data(), clut.mData.size());
142 return create_texture(id, core, clut);
143 } else {
144 return create_texture(id, core);
145 }
146}
