32void skybox_renderer::load_skybox(float size, int divisions)
33{
34 if(mDList) { glDeleteLists(mDList, 1); mDList = 0; }
35 if(!mDList) mDList = glGenLists(1);
36
37 float subSize = size / (float) divisions;
38
39 mDList = glGenLists(1);
40 glNewList(mDList, GL_COMPILE);
41 glBegin(GL_QUADS);
42 // chatgpt is the best, wrap shit in a dlist for moa pafomance
43 for (int face = 0; face < 6; ++face) {
44 for (int i = 0; i < 16; ++i) {
45 for (int j = 0; j < 16; ++j) {
46 float x = -size + j * subSize * 2;
47 float y = -size + i * subSize * 2;
48
49 // Texture coordinates for each sub-quad
50 float u = j / 16.0f;
51 float v = i / 16.0f;
52 float uNext = (j + 1) / 16.0f;
53 float vNext = (i + 1) / 16.0f;
54
55 // Calculate the vertices for each face based on the face index
56 float x0, y0, z0, x1, y1, z1, x2, y2, z2, x3, y3, z3;
57 switch (face) {
58 case 0: // Front face
59 x0 = x; y0 = y; z0 = -size;
60 x1 = x + 2*subSize; y1 = y; z1 = -size;
61 x2 = x + 2*subSize; y2 = y + 2*subSize; z2 = -size;
62 x3 = x; y3 = y + 2*subSize; z3 = -size;
63 break;
64 case 1: // Back face
65 x0 = x; y0 = y; z0 = size;
66 x1 = x + 2*subSize; y1 = y; z1 = size;
67 x2 = x + 2*subSize; y2 = y + 2*subSize; z2 = size;
68 x3 = x; y3 = y + 2*subSize; z3 = size;
69 break;
70 case 2: // Left face
71 x0 = -size; z0 = x; y0 = y;
72 x1 = -size; z1 = x + 2*subSize; y1 = y;
73 x2 = -size; z2 = x + 2*subSize; y2 = y + 2*subSize;
74 x3 = -size; z3 = x; y3 = y + 2*subSize;
75 break;
76 case 3: // Right face
77 x0 = size; z0 = x; y0 = y;
78 x1 = size; z1 = x + 2*subSize; y1 = y;
79 x2 = size; z2 = x + 2*subSize; y2 = y + 2*subSize;
80 x3 = size; z3 = x; y3 = y + 2*subSize;
81 break;
82 case 4: // Top face
83 x0 = x; y0 = size; z0 = y;
84 x1 = x + 2*subSize; y1 = size; z1 = y;
85 x2 = x + 2*subSize; y2 = size; z2 = y + 2*subSize;
86 x3 = x; y3 = size; z3 = y + 2*subSize;
87 break;
88 case 5: // Bottom face
89 x0 = x; y0 = -size; z0 = y;
90 x1 = x + 2*subSize; y1 = -size; z1 = y;
91 x2 = x + 2*subSize; y2 = -size; z2 = y + 2*subSize;
92 x3 = x; y3 = -size; z3 = y + 2*subSize;
93 break;
94 }
95
96 glTexCoord2f(u, v); glVertex3f(x0, y0, z0);
97 glTexCoord2f(uNext, v); glVertex3f(x1, y1, z1);
98 glTexCoord2f(uNext, vNext); glVertex3f(x2, y2, z2);
99 glTexCoord2f(u, vNext); glVertex3f(x3, y3, z3);
100 }
101 }
102 }
103 glEnd();
104 glEndList();
