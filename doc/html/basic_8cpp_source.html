143void basic_renderer::draw_arc(float x, float y, float width, float height, float angle1, float angle2, const math::color& color)
144{
145 float radius = (width + height) / 2;
146 int n = radius < 48 ? 12 : (radius / 4);
147 float m = 2.0f * math::PI / n;
148
149 float cx = x + width / 2.0F;
150 float cy = y + height / 2.0F;
151
152 int start = angle1 / 360.0F * n;
153 int end = angle2 / 360.0F * n;
154
155 glDisable(GL_TEXTURE_2D);
156 glBegin(GL_LINE_STRIP);
157 for(int i = start; i <= end; i++) {
158 glColor4fv(color.vector);
159 glVertex2f(
160 cx + width * math::cos(i * m) / 2.0F,
161 cy - height * math::sin(i * m) / 2.0F
162 );
163 }
164 glEnd();
165 glEnable(GL_TEXTURE_2D);
166}
