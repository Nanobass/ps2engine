30inline void
31Utils::MemCpy128(uint128_t* dest, const uint128_t* src, uint32_t numQwords)
32{
33 mAssert(((uint32_t)dest & 0xf) == 0 && ((uint32_t)src & 0xf) == 0);
34 while (numQwords-- > 0)
35 *(dest++) = *(src++);
36}
