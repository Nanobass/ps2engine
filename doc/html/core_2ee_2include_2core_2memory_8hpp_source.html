393struct default_allocator : public allocator
394{
395 void* allocate(size_t size) override
396 {
397 void* ptr = std::malloc(size);
398 size_t actual_size = malloc_usable_size(ptr);
399 mLargestBlock = std::max(mLargestBlock, actual_size);
400 mNumberOfAllocations++;
401 mUsed += actual_size;
402 mPeakUsage = std::max(mPeakUsage, mUsed);
403 return ptr;
404 }
405 void deallocate(void* ptr) override
406 {
407 size_t actual_size = malloc_usable_size(ptr);
408 mUsed -= actual_size;
409 mNumberOfAllocations--;
410 std::free(ptr);
411 }
412 bool allocated(void* ptr) override
413 {
414 // all non-definable allocations are managed by malloc
415 return true;
416 }
417 size_t get_allocation_size(void* ptr) override
418 {
419 return malloc_usable_size(ptr);
420 }
421 void get_statistics(size_t& total_free_memory, size_t& largest_block, size_t& number_of_allocations, size_t& peak_usage) override
422 {
423 total_free_memory = GetFreeSize();
424 largest_block = mLargestBlock;
425 number_of_allocations = mNumberOfAllocations;
426 peak_usage = mPeakUsage;
427 }
428 size_t get_size() override { return -1; };
429 allocator_id get_id() const override { return PSE_ALLOCATOR_DEFAULT; }
430 const char* get_name() const override { return "malloc"; }
431private:
432
433 void* AllocateLargestFreeBlock(size_t* Size)
434 {
435 size_t s0, s1;
436 void* p;
437
438 s0 = ~(size_t)0 ^ (~(size_t)0 >> 1);
439
440 while (s0 && (p = malloc(s0)) == NULL)
441 s0 >>= 1;
442
443 if (p)
444 free(p);
445
446 s1 = s0 >> 1;
447
448 while (s1)
449 {
450 if ((p = malloc(s0 + s1)) != NULL)
451 {
452 s0 += s1;
453 free(p);
454 }
455 s1 >>= 1;
456 }
457
458 while (s0 && (p = malloc(s0)) == NULL)
459 s0 ^= s0 & -s0;
460
461 *Size = s0;
462 return p;
463 }
464
465 size_t GetFreeSize(void)
466 {
467 size_t total = 0;
468 void* pFirst = NULL;
469 void* pLast = NULL;
470
471 for (;;)
472 {
473 size_t largest;
474 void* p = AllocateLargestFreeBlock(&largest);
475
476 if (largest < sizeof(void*))
477 {
478 if (p != NULL)
479 free(p);
480 break;
481 }
482
483 *(void**)p = NULL;
484
485 total += largest;
486
487 if (pFirst == NULL)
488 pFirst = p;
489
490 if (pLast != NULL)
491 *(void**)pLast = p;
492
493 pLast = p;
494 }
495
496 while (pFirst != NULL)
497 {
498 void* p = *(void**)pFirst;
499 free(pFirst);
500 pFirst = p;
501 }
502
503 return total;
504 }
505
506 size_t mLargestBlock = 0;
507 size_t mNumberOfAllocations = 0;
508 size_t mUsed = 0;
509 size_t mPeakUsage = 0;
