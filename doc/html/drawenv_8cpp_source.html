76void CDrawEnv::InitCommon(GS::tContext context)
77{
78 // set up the dma end tag
79 SettingsDmaTag.QWC = uiNumGSRegs + 1; // num regs + 1 giftag
80 SettingsDmaTag.PCE = 0;
81 SettingsDmaTag.ID = DMAC::kEnd;
82 SettingsDmaTag.IRQ = 0; // no irq
83 SettingsDmaTag.ADDR = 0; // no next tag
84 SettingsDmaTag.SPR = 0; // not from sp
85
86 // set up the gif tag
87 // zero it first
88
89 SettingsGifTag.NLOOP = uiNumGSRegs;
90 SettingsGifTag.EOP = 1;
91 SettingsGifTag.PRE = 0;
92 SettingsGifTag.FLG = 0;
93 SettingsGifTag.NREG = 1;
94 SettingsGifTag.REGS0 = 0xe;
95
96 eZTestPassMode = ZTest::kGEqual;
97 uiFBWidth = 0;
98 uiFBHeight = 0;
99 InterlacedOffset = 0.5f;
100
101 // first set addresses for the registers that are context-independent
102 PrModeContAddr = GS::RegAddrs::prmodecont;
103 ColClampAddr = GS::RegAddrs::colclamp;
104 PABEAddr = GS::RegAddrs::pabe;
105 DTHEAddr = GS::RegAddrs::dthe;
106 DIMXAddr = GS::RegAddrs::dimx;
107 FogColAddr = GS::RegAddrs::fogcol;
108
109 SetContext(context);
110
111 gsrPrModeCont.control = 1; // use PRIM register
112 EnableColorClamp();
113 DisableSelectiveAlphaBlend();
114 SetFrameBufAlphaORMask(0);
115 DisableDither();
116 using namespace GS::Dither;
117 SetDitherMatrix(kMinusFour, kTwo, kMinusThree, kThree,
118 kZero, kMinusTwo, kOne, kMinusOne,
119 kMinusThree, kThree, kMinusFour, kTwo,
120 kOne, kMinusOne, kZero, kMinusTwo);
121 SetFogColor(0, 0, 0);
122
123 // gsrFrame.FBP = 0x0; // just to have something...
124 // gsrFrame.FBW = 0; // invalid
125 SetFrameBufferPSM(GS::kPsm32);
126 SetFrameBufferDrawMask(0); // enable write to frame buffer
127
128 gsrZBuf.fb_addr = 0; // after two 640x224 frame buffers
129 gsrZBuf.psm = 0; // KPSMZ32
130 gsrZBuf.update_mask = 0; // enable write to z-buffer
131
132 gsrXYOffset.offset_x = 0; // probably won't make a lot of sense later...
133 gsrXYOffset.offset_y = 0;
134
135 gsrScissor.clip_x0 = 0;
136 gsrScissor.clip_y0 = 0;
137 gsrScissor.clip_x1 = 0x7ff;
138 gsrScissor.clip_y1 = 0x7ff;
139
140 gsrTest.atest_enable = 0; // disable alpha test
141 gsrTest.atest_method = 0; // never
142 gsrTest.atest_reference = 0x80; // identity
143 gsrTest.atest_fail_method = 0; // keep - ?
144 gsrTest.datest_enable = 0; // disable destination alpha test
145 gsrTest.datest_mode = 0; // 0 = pass
146 gsrTest.ztest_enable = 1; // enable z-test
147 gsrTest.ztest_method = (uint64_t)eZTestPassMode;
148
149 // make sure things are qword aligned (I don't trust the compiler...)
150 mAssert(((uint32_t)&SettingsGifTag & 0xf) == 0);
151}

153void CDrawEnv::CalculateClippedFBXYOffsets(bool addHalfPixel)
154{
155 float gsCenterX, gsCenterY;
156 gsCenterX = gsCenterY = 4096.0f / 2.0f;
157
158 float gsOffsetX, gsOffsetY;
159 gsOffsetX = gsCenterX - (float)(uiFBWidth / 2);
160 gsOffsetY = gsCenterY - (float)(uiFBHeight / 2);
161
162 if (addHalfPixel)
163 gsOffsetY += InterlacedOffset;
164
165 gsrXYOffset.offset_x = (uint16_t)Core::FToI4(gsOffsetX);
166 gsrXYOffset.offset_y = (uint16_t)Core::FToI4(gsOffsetY);
167}

174void CDrawEnv::SendSettings(CSCDmaPacket& packet)
175{
176 bool opened_tag;
177 if ((opened_tag = !packet.HasOpenTag()))
178 packet.Cnt();
179 packet.Add(&SettingsGifTag, uiNumGSRegs + 1);
180 if (opened_tag)
181 packet.CloseTag();
182}

184void CDrawEnv::SendSettings(CVifSCDmaPacket& packet)
185{
186 // the data needs to be qword-aligned, so pad with appropriate # of vifnops to
187 // put the direct vifcode at the end of a qword
188 bool opened_tag = false;
189 if (packet.HasOpenTag()) {
190 // assume that the packet is aligned to 128 bits
191 packet.Nop().Nop().Nop();
192 } else {
193 opened_tag = true;
194 packet.Cnt();
195 packet.Nop();
196 if (!packet.GetTTE()) {
197 packet.Nop().Nop();
198 }
199 }
200
201 packet.OpenDirect();
202 packet.Add(&SettingsGifTag, uiNumGSRegs + 1);
203 packet.CloseDirect();
204
205 if (opened_tag)
206 packet.CloseTag();
207}

209void CDrawEnv::SetFrameBufferDim(uint32_t pixelW, uint32_t pixelH)
210{
211 // width must be a multiple of 64
212 if (pixelW < 64)
213 pixelW = 64;
214 mAssert((pixelW & 63) == 0);
215
216 gsrFrame.fb_width = pixelW / 64;
217 uiFBWidth = pixelW;
218 uiFBHeight = pixelH;
219
220 gsrScissor.clip_x0 = 0;
221 gsrScissor.clip_y0 = 0;
222 gsrScissor.clip_x1 = pixelW - 1;
223 gsrScissor.clip_y1 = pixelH - 1;
224}
