43class padman_device : public input_device
44{
45public:
46 padman_device(int port, int slot = 0)
47 : mPort(port), mSlot(slot)
48 {
49 mName = std::string("padman_device: ") + (char)('0' + mPort) + (char)('A' + mSlot);
50
51 if(!padPortOpen(mPort, mSlot, mPadBuffer)) throw padman_exception(kNoDevice, mPort, mSlot);
52 sync();
53
54 // check if controller has a dualshock mode
55 int i = 0, modes = padInfoMode(mPort, mSlot, PAD_MODETABLE, -1);
56 for(;i < modes; i++) {
57 if (padInfoMode(mPort, mSlot, PAD_MODETABLE, i) == PAD_TYPE_DUALSHOCK) break;
58 }
59 if(i >= modes) throw padman_exception(kInvalidDevice, mPort, mSlot);
60
61 // do some more checking
62 if(!padInfoMode(mPort, mSlot, PAD_MODECUREXID, 0)) throw padman_exception(kInvalidDevice, mPort, mSlot);
63
64 // lock the mode button
65 padSetMainMode(mPort, mSlot, PAD_MMODE_DUALSHOCK, PAD_MMODE_LOCK);
66 sync();
67
68 mActuatorCount = padInfoAct(mPort, mSlot, -1, 0);
69 if(mActuatorCount != 0) {
70 mActAlign[0] = 0; // Enable small engine
71 mActAlign[1] = 1; // Enable big engine
72 mActAlign[2] = 0xff;
73 mActAlign[3] = 0xff;
74 mActAlign[4] = 0xff;
75 mActAlign[5] = 0xff;
76 padSetActAlign(mPort, mSlot, mActAlign);
77 sync();
78 }

116 void sync()
117 {
118 int state = -1;
119 int lastState = -1;
120 do {
121 state = padGetState(mPort, mSlot);
122 if (state != lastState) {
123 char stateString[16];
124 padStateInt2String(state, stateString);
125 pse::log::out(pse::log::kDebug) << mName << " state='" << stateString << "'" << std::endl;
126 }
127 if (state == PAD_STATE_DISCONN) throw padman_exception(kDisconnected, mPort, mSlot);
128 if (state == PAD_STATE_ERROR) throw padman_exception(kDriverError, mPort, mSlot);
129 lastState = state;
130 } while((state != PAD_STATE_STABLE) && (state != PAD_STATE_FINDCTP1));
131 if (lastState == -1) throw padman_exception(kSyncError, mPort, mSlot);
132 }
