49void initialize_memory_system()
50{
51 if(g_MemorySystemInitialized) return;
52 pse::memory::initialize();
53 pse::memory::register_douglea_allocator(GME_OBJECT, pse::memory::PSE_ALLOCATOR_DEFAULT, "Object Heap", 1024 * 1024 * 2);
54 g_MemorySystemInitialized = true;
55 std::atexit(exit_handler);
56 std::set_terminate(crash_handler);
57}
