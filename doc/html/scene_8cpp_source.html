45game_object scene::create_entity_with_uuid(uuid uuid, const std::string& name)
46{
47 memory::push_allocator(GME_OBJECT);
48 game_object entity = { mRegistry.create(), this };
49 entity.add_component<id_component>(uuid);
50 entity.add_component<transform_component>();
51 auto& tag = entity.add_component<tag_component>();
52 tag.mTag = name.empty() ? "game_object" : name;
53 mEntityMap[uuid] = entity;
54 memory::pop_allocator();
55 return entity;
56}

97void scene::render()
98{
99 transform_component* trCamera = nullptr;
100 camera_component* gcCamera = nullptr;
101
102 {
103 auto view = mRegistry.view<transform_component, camera_component>();
104 for(auto entity : view)
105 {
106 auto [transform, camera] = view.get<transform_component, camera_component>(entity);
107 if(camera.mPrimary)
108 {
109 trCamera = &transform;
110 gcCamera = &camera;
111 break;
112 }
113 }
114 }
115
116 if(trCamera && gcCamera)
117 {
118 // TODO: optimize!!!
119 glMatrixMode(GL_PROJECTION);
120 gcCamera->mCamera.set_viewport(mRenderManager->mWidth, mRenderManager->mHeight, mRenderManager->mAspectRatio);
121 gcCamera->mCamera.apply_projection();
122 glMatrixMode(GL_MODELVIEW);
123 glLoadIdentity();
124 glScalef(1.0F / trCamera->mScale.x, 1.0F / trCamera->mScale.y, 1.0F / trCamera->mScale.z);
125 glRotatef(-trCamera->mRotation.x, 1.0F, 0.0F, 0.0F);
126 glRotatef(-trCamera->mRotation.y, 0.0F, 1.0F, 0.0F);
127 glRotatef(-trCamera->mRotation.z, 0.0F, 0.0F, 1.0F);
128 glTranslatef(-trCamera->mTranslation.x, -trCamera->mTranslation.y, -trCamera->mTranslation.z);
129 //math::mat4 viewMatrix = trCamera->get_transform();
130 //pglLoadMatrixCombined(viewMatrix.inverted(), viewMatrix);
131 }
132 else log::out(log::kError) << "no camera present" << std::endl;
133
134 mRenderManager->mLightingManager->set_lighting_enabled(false);
135
136 if(trCamera && gcCamera)
137 {
138 auto view = mRegistry.view<transform_component, skybox_component>();
139 for(auto& entity : view)
140 {
141 auto [transform, skybox] = view.get<transform_component, skybox_component>(entity);
142 if(!skybox.mSkyTexture) continue;
143 mRenderManager->mSkyboxRenderer->set_sky_texture(skybox.mSkyTexture);
144 mRenderManager->mSkyboxRenderer->render_sky(trCamera->mTranslation);
145 }
146 } else glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);
147
148 mRenderManager->mLightingManager->set_lighting_enabled(true);
149
150 {
151 auto view = mRegistry.view<transform_component, light_component>();
152 for(auto& entity : view)
153 {
154 auto [transform, light] = view.get<transform_component, light_component>(entity);
155 if(!light.mLight) continue;
156 switch(light.mLightType)
157 {
158 case light_component::kPosition:
159 light.mLight->set_position(math::vec4(-transform.mTranslation, 1.0F));
160 break;
161 case light_component::kDirectional:
162 {
163 math::mat4 matrix = math::rotation(transform.mRotation);
164 math::vec4 direction = matrix * math::vec4(0.0F, 0.0F, 1.0F, 0.0F);
165 light.mLight->set_position(math::vec4(direction, 0.0F));
166 }
167 break;
168 default:
169 break;
170 }
171 light.mLight->set_enabled(true);
172 light.mLight->apply();
173 }
174 }
175
176 {
177 auto view = mRegistry.view<transform_component, mesh_renderer_component>();
178 for(auto& entity : view)
179 {
180 auto [transform, mesh] = view.get<transform_component, mesh_renderer_component>(entity);
181 if(!mesh.mMesh) continue;
182 glPushMatrix();
183 pglMultMatrix(transform.get_transform());
184 for(auto& sub : mesh.mMesh->mSubMeshes)
185 {
186 sub.draw_arrays();
187 }
188 glPopMatrix();
189 }
190 }
191
192 {
193 auto view = mRegistry.view<transform_component, sprite_renderer_component>();
194 for(auto& entity : view)
195 {
196 auto [transform, sprite] = view.get<transform_component, sprite_renderer_component>(entity);
197 if(!sprite.mTexture) continue;
198 glPushMatrix();
199 pglMultMatrix(transform.get_transform());
200 mRenderManager->mSpriteRenderer->render_sprite(sprite.mTexture, sprite.mOffset, sprite.mScale, sprite.mColor);
201 glPopMatrix();
202 }
203 }
