43void CDisplayEnv::SetFB(tGsrDispFB* dispFBReg,
44 unsigned int wordAddr, unsigned int bufWidth,
45 unsigned int xOffsetInBuf, unsigned int yOffsetInBuf,
46 unsigned int psm)
47{
48 mErrorIf(wordAddr & 2047, "Frame buffers must be page aligned (8k)");
49
50 dispFBReg->FBP = wordAddr / 2048; // page aligned
51 dispFBReg->FBW = bufWidth / 64; // width = num pixels / 64
52 dispFBReg->PSM = psm;
53 dispFBReg->DBX = xOffsetInBuf; // pixel offset
54 dispFBReg->DBY = yOffsetInBuf; // pixel offset
55}

57void CDisplayEnv::SetDisplay(tGsrDisplay* displayReg,
58 unsigned int width, unsigned int height,
59 unsigned int screenX, unsigned int screenY,
60 unsigned int magH, unsigned int magV)
61{
62 mErrorIf(magH == 0 || magV == 0, "magH and magV can't be zero.");
63
64 // displayReg->DX = 302;
65 // displayReg->DY = 24;
66 // displayReg->MAGH = 0;
67 // displayReg->MAGV = 0;
68 // displayReg->DW = 1279;
69 // displayReg->DH = 719;
70
71 // printf("this = %p, displayReg = %p\n", this, displayReg );
72
73 displayReg->DX = HorizontalOverScan + screenX * magH;
74 displayReg->DY = VerticalOverScan + screenY * magV;
75 displayReg->MAGH = magH - 1;
76 displayReg->MAGV = magV - 1;
77 displayReg->DW = width * magH; // width = num pixels * clocks per pixel
78 displayReg->DH = (height - 1) * magV;
