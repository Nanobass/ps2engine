63void CTexEnv::InitCommon(GS::tContext context)
64{
65 mAssert(context == GS::kContext1 || context == GS::kContext2);
66
67 uiTexPixelWidth = uiTexPixelHeight = 0;
68
69 // set up the settings xfer dma tag
70 SettingsDmaTag.QWC = uiNumSettingsGSRegs + 1; // num regs + 1 giftag
71 SettingsDmaTag.PCE = 0;
72 SettingsDmaTag.ID = DMAC::kEnd;
73 SettingsDmaTag.IRQ = 0; // no irq
74 SettingsDmaTag.ADDR = 0; // no next tag
75 SettingsDmaTag.SPR = 0; // not from sp
76
77 // setup the texturing settings gif tag
78 SettingsGifTag.NLOOP = uiNumSettingsGSRegs;
79 SettingsGifTag.EOP = 1;
80 SettingsGifTag.PRE = 0;
81 SettingsGifTag.FLG = 0; // packed
82 SettingsGifTag.NREG = 1;
83 SettingsGifTag.REGS0 = 0xe;
84
85 // register addresses
86
87 // first the addresses for the registers that are context-independent
88 // flush by default
89 TexflushAddr = GS::RegAddrs::texflush;
90 TexAAddr = GS::RegAddrs::texa;
91
92 SetContext(context); // do the context-dependent stuff
93
94 // gs register defaults
95
96 gsrTexA.alpha_0 = 0x80; // identity
97 gsrTexA.alpha_1 = 0x80; // identity
98 gsrTexA.alpha_method = 0;
99
100 // don't use SetWrapMode() here!
101 // make this default to clamp because I want to issue an error when
102 // SetRegion is called if the mode is repeat and the width or height is not
103 // of the form (2^x -1). If I start this off as repeat and the user wants to
104 // use clamp mode then they will have to call SetWrapMode before SetRegion, which
105 // isn't good.
106 gsrClamp.wrap_mode_s = TexWrapMode::kClamp;
107 gsrClamp.wrap_mode_t = TexWrapMode::kClamp;
108
109 // gsrTex0.tb_addr = 0;
110 // gsrTex0.tb_width = 0x3f; // invalid setting
111 gsrTex0.psm = GS::kPsm32;
112 // gsrTex0.tex_width = 0xf; // invalid setting
113 // gsrTex0.tex_height = 0xf; // invalid setting
114 gsrTex0.tex_cc = 1; // use texture alpha
115 gsrTex0.tex_funtion = TexMode::kModulate;
116 // gsrTex0.cb_addr = 0x3fff;
117 // gsrTex0.clut_pixmode = SCE_GS_PSMCT32;
118 gsrTex0.clut_smode = 0; // csm1 storage mode (don't reference texclut)
119 gsrTex0.clut_offset = 0; // talk about a weird addressing mode... I'm not even going to mess with this now...
120 gsrTex0.clut_loadmode = 4; // load clut to buffer if cbp0 != cbp
121
122 gsrTex1.lcm = 1; // fixed LOD
123 gsrTex1.mxl = 0; // max mip map level
124 SetMagMode(MagMode::kLinear);
125 gsrTex1.mmin = 1; // linear
126 gsrTex1.mtba = 0; // use miptbp1/2
127 gsrTex1.l = 0;
128 gsrTex1.k = 0;
129
130 // make sure things are qword aligned (I don't trust the compiler...)
131 mAssert(((uint32_t)&SettingsGifTag & 0xf) == 0);
132}

197void CTexEnv::SetDimensions(uint32_t w, uint32_t h)
198{
199 uiTexPixelWidth = w;
200 uiTexPixelHeight = h;
201
202 // get the log base 2 of the dimensions
203 uint32_t logW = Math::Log2(w);
204 uint32_t logH = Math::Log2(h);
205
206 // if the texture dimensions are not powers of two, they need to be rounded up
207 // to the next power of two
208 if (((uint32_t)1 << logW) != w)
209 logW++;
210 if (((uint32_t)1 << logH) != h)
211 logH++;
212
213 gsrTex0.tex_width = logW;
214 gsrTex0.tex_height = logH;
215
216 // the width of the texture buffer should be a multiple of the page size for
217 // perspective tex mapping (but what about plain old (u,v)?? -- FIXME)
218
219 int page_width = 64;
220 unsigned int bpp = GS::GetBitsPerPixel((GS::tPSM)gsrTex0.psm);
221 if (bpp == 8 || bpp == 4)
222 page_width = 128;
223
224 gsrTex0.tb_width = ((w + page_width - 1) / page_width) * page_width / 64;
225
226 // if we're using bi- or tri-linear filtering the tex needs to be at least 8x8
227 mAssert((gsrTex1.mmag == MagMode::kNearest) || (logW >= 3 && logH >= 3));
228 mAssert((gsrTex1.mmin == MinMode::kNearest) || (gsrTex1.mmin == MinMode::kNearestMipmapNearest) || (logW >= 3 && logH >= 3));
229}

231// Caution about regions: Note that in 3.4.5 "Texture Wrap Modes" REPEAT, CLAMP, and REGION_CLAMP, are
232// all pretty reasonable, but REGION_REPEAT uses the bitwise operations & and | instead of the % and + operations
233// that you would expect. RTM.
234
235void CTexEnv::SetRegion(uint32_t originU, uint32_t originV, uint32_t w, uint32_t h)
236{
237 tTexWrapMode sMode, tMode;
238 sMode = (gsrClamp.wrap_mode_s & 2) ? (tTexWrapMode)(3 - gsrClamp.wrap_mode_s) : (tTexWrapMode)gsrClamp.wrap_mode_s;
239 tMode = (gsrClamp.wrap_mode_t & 2) ? (tTexWrapMode)(3 - gsrClamp.wrap_mode_t) : (tTexWrapMode)gsrClamp.wrap_mode_t;
240
241 // the meaning of MIN/MAX in gsrClamp differs in clamp and repeat modes
242
243 using namespace TexWrapMode;
244 if (sMode == kClamp) {
245 gsrClamp.min_clamp_u = originU;
246 gsrClamp.max_clamp_u = originU + w - 1;
247 } else {
248 gsrClamp.min_clamp_u = w - 1;
249 gsrClamp.max_clamp_u = originU;
250 }
251
252 if (tMode == kClamp) {
253 gsrClamp.min_clamp_v = originV;
254 gsrClamp.max_clamp_v = originV + h - 1;
255 } else {
256 gsrClamp.min_clamp_v = h - 1;
257 gsrClamp.max_clamp_v = originV;
258 }
259
260 gsrClamp.wrap_mode_s = 3 - (uint32_t)sMode;
261 gsrClamp.wrap_mode_t = 3 - (uint32_t)tMode;
262
263 // see note above this method
264 mAssert((sMode == kClamp) || (((gsrClamp.max_clamp_u & gsrClamp.min_clamp_u) == 0) && Math::IsPow2(gsrClamp.min_clamp_u + 1)));
265 mAssert((tMode == kClamp) || (((gsrClamp.max_clamp_v & gsrClamp.min_clamp_v) == 0) && Math::IsPow2(gsrClamp.min_clamp_v + 1)));
266}

287void CTexEnv::SetWrapModeS(tTexWrapMode sMode)
288{
289 using namespace TexWrapMode;
290 // the meaning of MIN/MAX differs between clamp and wrap modes,
291 // so if we're using a region and
292 // the old mode is not the new mode we need to convert the region data
293 tTexWrapMode oldMode = (gsrClamp.wrap_mode_s & 2) ? (tTexWrapMode)(3 - gsrClamp.wrap_mode_s) : (tTexWrapMode)sMode;
294 if (oldMode != sMode) {
295 if (oldMode == kRepeat) {
296 uint32_t temp = gsrClamp.min_clamp_u;
297 gsrClamp.min_clamp_u = gsrClamp.max_clamp_u;
298 gsrClamp.max_clamp_u += temp;
299 } else {
300 uint32_t temp = gsrClamp.min_clamp_u;
301 gsrClamp.min_clamp_u = gsrClamp.max_clamp_u - gsrClamp.min_clamp_u;
302 gsrClamp.max_clamp_u = temp;
303 }
304 }
305 // are we using a region?
306 if (gsrClamp.wrap_mode_s & 2)
307 sMode = (tTexWrapMode)(3 - sMode);
308
309 gsrClamp.wrap_mode_s = sMode;
310
311 // see note above SetRegion()
312 mAssert((gsrClamp.wrap_mode_s <= 2) || (((gsrClamp.max_clamp_u & gsrClamp.min_clamp_u) == 0) && Math::IsPow2(gsrClamp.min_clamp_u + 1)));
313}

397void CTexture::SetImage(uint128_t* imagePtr, uint32_t w, uint32_t h, GS::tPSM psm, uint32_t* clutPtr)
398{
399 // make sure the image is qword aligned
400 mAssert(((uint32_t)imagePtr & 0xf) == 0);
401
402 pImage = imagePtr;
403
404 CTexEnv::SetPSM(psm);
405
406 // dimensions
407 CTexEnv::SetDimensions(w, h);
408
409 // I think this isn't necessary anymore...
410 /*
411 // width of the area in gs mem (64 pixel units) to use
412 uint32_t gsBufWidth = ( (w % 64) != 0 ) ? w/64 + 1 : w/64;
413 gsBufWidth = Math::Max( ((uint32_t)1 << gsrTex0.tex_width) / 64, gsBufWidth );
414 gsrTex0.tb_width = gsBufWidth;
415 */
416
417 unsigned int gsBufWidth = gsrTex0.tb_width * 64;
418 pImageUploadPkt->SetGsBufferWidth(gsBufWidth);
419 // width and height of the region to xfer
420 pImageUploadPkt->SetImage(imagePtr, w, h, psm);
