49void input_manager::detect_controllers()
50{
51 bool padmanActivePads[8];
52 for(auto& device : mDevices) {
53 if(padman_device* padmanDevice = dynamic_cast<padman_device*>(device.get())) {
54 padmanActivePads[padmanDevice->mSlot*4+padmanDevice->mPort] = true;
55 }
56 }
57 for(int i = 0; i < 1; i++) {
58 if(!padmanActivePads[i]) {
59 try {
60 padman_device* padmanDevice = new padman_device(i / 4, i % 4);
61 mDevices.emplace_back(padmanDevice);
62 } catch(const input_exception& ie) {
63 switch (ie.mInputError)
64 {
65 case kDisconnected:
66 case kNoDevice:
67 case kInvalidDevice:
68 case kSyncError:
69 default:
70 break;
71 }
72 }
73 }
74 }
75}
