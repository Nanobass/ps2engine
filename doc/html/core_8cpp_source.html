51void SetupPerfCounters(uint32_t evt_0, uint32_t evt_1)
52{
53 asm __volatile__(
54
55 ".set noreorder \n"
56
57 "mtps $0,0 # halt performance counters \n"
58 "sync.p # \n"
59
60 "mtpc $0,0 # set perfcounter 0 to $0 \n"
61 "mtpc $0,1 # set perfcounter 1 to $0 \n"
62
63 "sync.p # \n"
64
65 "andi %0,%0,31 # mask off unused bits \n"
66 "andi %1,%1,31 \n"
67
68 "sll %0,%0,5 \n"
69 "sll %1,%1,15 \n"
70
71 "ori %0,%0,16 # set U0 \n"
72 "ori %1,%1,1<<14 # set U1 \n"
73
74 "or %0,%1,%0 \n"
75 "lui %1,0x8000 # master enable \n"
76
77 "or %0,%1,%0 # set top bit - this is Counter Enable \n"
78 "mtps %0,0 # truly - we rule \n"
79 "sync.p \n"
80
81 ".set reorder \n"
82
83 : // no output
84 : "r"(evt_0), "r"(evt_1));
85}
