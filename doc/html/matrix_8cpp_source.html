150void glFrustum(GLdouble left, GLdouble right,
151 GLdouble bottom, GLdouble top,
152 GLdouble zNear, GLdouble zFar)
153{
154 /*
155 * NOTE:
156 * The PS2 does not support GL_LESS/GL_LEQUAL
157 * but this is what 99% of OpenGL programs use.
158 *
159 * As a result depth is inverted.
160 * See glDepthFunc/glFrustum/glOrtho
161 */
162 GL_FUNC_DEBUG("%s\n", __FUNCTION__);
163
164 pse::math::mat4 xform(
165 (2.0f * zNear) / (right - left),
166 0.0f,
167 0.0f,
168 0.0f,
169 0.0f,
170 (2.0f * zNear) / (top - bottom),
171 0.0f,
172 0.0f,
173 (right + left) / (right - left),
174 (top + bottom) / (top - bottom),
175 -(zFar + zNear) / (zFar - zNear),
176 -1.0f,
177 0.0f,
178 0.0f,
179 (-2.0f * zFar * zNear) / (zFar - zNear),
180 0.0f
181 );
182
183 pse::math::mat4 inv(
184 (right - left) / (2 * zNear),
185 0,
186 0,
187 0,
188 0,
189 (top - bottom) / (2 * zNear),
190 0,
191 0,
192 0,
193 0,
194 0,
195 -(zFar - zNear) / (2 * zFar * zNear),
196 (right + left) / (2 * zNear),
197 (top + bottom) / (2 * zNear),
198 -1,
199 (zFar + zNear) / (2 * zFar * zNear)
200 );
201
202 CMatrixStack& matStack = pGLContext->GetCurMatrixStack();
203 matStack.Concat(xform, inv);
204}

206void glOrtho(GLdouble left, GLdouble right,
207 GLdouble bottom, GLdouble top,
208 GLdouble zNear, GLdouble zFar)
209{
210 /*
211 * NOTE:
212 * The PS2 does not support GL_LESS/GL_LEQUAL
213 * but this is what 99% of OpenGL programs use.
214 *
215 * As a result depth is inverted.
216 * See glDepthFunc/glFrustum/glOrtho
217 */
218 GL_FUNC_DEBUG("%s\n", __FUNCTION__);
219
220 pse::math::mat4 xform(
221 (2.0f) / (right - left),
222 0.0f,
223 0.0f,
224 0.0f,
225 0.0f,
226 (2.0f) / (top - bottom),
227 0.0f,
228 0.0f,
229 0.0f,
230 0.0f,
231 -2 / (zFar - zNear),
232 0.0f,
233 -(right + left) / (right - left),
234 -(top + bottom) / (top - bottom),
235 -(zFar + zNear) / (zFar - zNear),
236 1.0f
237 );
238
239 pse::math::mat4 inv(
240 (right - left) / 2,
241 0,
242 0,
243 0,
244 0,
245 (top - bottom) / 2,
246 0,
247 0,
248 0,
249 0,
250 (zFar - zNear) / -2,
251 0,
252 (right + left) / 2,
253 (top + bottom) / 2,
254 (zFar + zNear) / 2,
255 1
256 );
257
258 CMatrixStack& matStack = pGLContext->GetCurMatrixStack();
259 matStack.Concat(xform, inv);
