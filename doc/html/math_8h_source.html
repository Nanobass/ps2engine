96 " .set noreorder \n"

98 " cvt.w.s %0, %1 /* ret = (int)a */ \n"
99 " nop \n"

102 " nop \n"
103
104 " cvt.s.w %0, %0 /* ret = (float)ret */ \n"
105 " nop \n"

108 " nop \n"

119 " sub.s %0, %0, %1 \n"

123 " nop \n"

128 : "=&f"(ret)
129 : "f"(a), "r"(a)
