118void text_renderer::draw_string(font_ptr fnt, const std::string& string)
119{
120 float x = 0.0F, y = 0.0F;
121
122 glDisable(GL_DEPTH_TEST);
123
124 fnt->mFontTexture->bind();
125
126 glBegin(GL_QUADS);
127 for(auto chr : string)
128 {
129 glyph glyph = fnt->mGlyphs[chr];
130
131 if(chr == '\n')
132 {
133 y -= 1.0F;
134 x = 0.0F;
135 }
136
137 float x1 = x + glyph.mXOffset;
138 float y1 = y + glyph.mYOffset + glyph.mHeight;
139 float x2 = x + glyph.mXOffset + glyph.mWidth;
140 float y2 = y + glyph.mYOffset;
141
142 glTexCoord2f(glyph.mU1, glyph.mV1);
143 glVertex2f(x1, y1); // TL
144 glTexCoord2f(glyph.mU2, glyph.mV1);
145 glVertex2f(x2, y1); // TR
146 glTexCoord2f(glyph.mU2, glyph.mV2);
147 glVertex2f(x2, y2); // BR
148 glTexCoord2f(glyph.mU1, glyph.mV2);
149 glVertex2f(x1, y2); // BL
150
151 x += glyph.mXAdvance;
152 }
153 glEnd();
154}
