105class CDListCmdBlock {
106 char Memory[kDListByteSize];
107 char* MemCursor;
108 int BytesLeft;
109 CDListCmdBlock* NextBlock;
110
111public:
112 CDListCmdBlock()
113 : MemCursor(Memory)
114 , BytesLeft(kDListByteSize)
115 , NextBlock(NULL)
116 {
117 CEmptyListCmd empty;
118 memcpy(MemCursor, &empty, CDListCmd::SizeOf<CEmptyListCmd>());
119 }
120 ~CDListCmdBlock()
121 {
122 if (NextBlock)
123 delete NextBlock;
124 }
125
126 template <class CmdType>
127 bool CanFit(CmdType cmd)
128 {
129 return (CDListCmd::SizeOf<CmdType>()
130 <= BytesLeft - CDListCmd::SizeOf<CNextBlockCmd>());
131 }
132
133 template <class CmdType>
134 void operator+=(CmdType cmd)
135 {
136 memcpy(MemCursor, &cmd, sizeof(CmdType)); // this should be the usual sizeof()
137 MemCursor += CDListCmd::SizeOf<CmdType>();
138 BytesLeft -= CDListCmd::SizeOf<CmdType>();
139 }
140
141 CDListCmd* GetFirstCmd() { return reinterpret_cast<CDListCmd*>(&Memory[0]); }
142
143 void SetNextBlock(CDListCmdBlock* next) { NextBlock = next; }
144 CDListCmdBlock* GetNextBlock() const { return NextBlock; }
145};
146
147inline CDListCmd*
148CNextBlockCmd::Play()
149{
150 return NextBlock->GetFirstCmd();
151}

157class CDList {
158 CDmaPacket *VertexBuf, *NormalBuf, *TexCoordBuf, *ColorBuf;
159 CDListCmdBlock *FirstCmdBlock, *CurCmdBlock;
160 int NumRenderPackets;
161 CVifSCDmaPacket* RenderPackets[kDListMaxNumRenderPackets];
162
163public:
164 CDList();
165 ~CDList();
166
167 // utility
168 template <class CmdType>
169 CDListCmd* GetNext(CmdType* cmd)
170 {
171 return 0;
172 }
173
174 template <class CmdType>
175 void operator+=(CmdType cmd)
176 {
177 if (!CurCmdBlock->CanFit(cmd)) {
178 // not enough space left in current cmd block, so terminate it
179 // and start a new one..
180 CDListCmdBlock* newBlock = new CDListCmdBlock;
181 CurCmdBlock->SetNextBlock(newBlock);
182 *CurCmdBlock += CNextBlockCmd(newBlock);
183 CurCmdBlock = newBlock;
184 }
185 *CurCmdBlock += cmd;
186 }
187
188 void Play()
189 {
190 CDListCmd* nextCmd = FirstCmdBlock->GetFirstCmd();
191 while (nextCmd)
192 nextCmd = nextCmd->Play();
193 }
194
195 void Begin();
196 void End()
197 {
198 CEndListCmd endList;
199 *this += endList;
200 }
201
202 CDmaPacket& GetVertexBuf();
203 CDmaPacket& GetNormalBuf();
204 CDmaPacket& GetTexCoordBuf();
205 CDmaPacket& GetColorBuf();
206
207 void RegisterNewPacket(CVifSCDmaPacket* packet)
208 {
209 RenderPackets[NumRenderPackets++] = packet;
210 }
211};
