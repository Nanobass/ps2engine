207inline unsigned int
208GS::GetBitsPerPixel(tPSM psm)
209{
210 uint32_t bpp = 0;
211
212 switch (psm) {
213 case kPsm32:
214 case kPsmz32:
215 bpp = 32;
216 break;
217 case kPsm24:
218 case kPsmz24:
219 bpp = 24;
220 break;
221 case kPsm16:
222 case kPsm16s:
223 case kPsmz16:
224 case kPsmz16s:
225 bpp = 16;
226 break;
227 case kPsm8:
228 case kPsm8h:
229 bpp = 8;
230 break;
231 case kPsm4:
232 case kPsm4hl:
233 case kPsm4hh:
234 bpp = 4;
235 break;
236 default:
237 mAssert(false);
238 break;
239 }
240
241 return bpp;
242}
