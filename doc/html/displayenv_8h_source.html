157void CDisplayEnv::SetDisplayMode(tDisplayMode mode)
158{
159 using namespace DisplayModes;
160
161 switch (mode) {
162 case ntsc:
163 HorizontalOverScan = 158 * 4;
164 VerticalOverScan = 50 * 1;
165 break;
166 case pal:
167 HorizontalOverScan = 158 * 4; // untested!
168 VerticalOverScan = 50 * 1; // untested!
169 break;
170 case dtv:
171 HorizontalOverScan = 302;
172 VerticalOverScan = 24;
173 break;
174 default:
175 mError("This display mode isn't handled yet..somebody type in the values.");
176 return;
177 }
178}
