110void* allocate(allocator_id alloc, size_t size)
111{
112 check_allocator(alloc);
113 allocator* allocator_ptr = g_Allocators[alloc];
114 void* ptr = allocator_ptr->allocate(size);
115 if (!ptr)
116 {
117 out_of_memory_callback(alloc, size);
118 throw std::bad_alloc();
119 }
120 size_t actual_size = allocator_ptr->get_allocation_size(ptr);
121 if(g_TrackingEnabled)
122 {
123 pse::perfmon::messages::allocation msg(alloc, actual_size, ptr, g_TrackingId);
124 pse::perfmon::post_message(msg, sizeof(msg));
125 }
126 return ptr;
127}

172allocator_id find_allocator(void* ptr)
173{
174 for (int i = 1; i < g_MaxAllocators; ++i)
175 {
176 if (g_Allocators[i] && g_Allocators[i]->allocated(ptr)) return static_cast<allocator_id>(i);
177 }
178 // unknown memory must have been malloc't by the system
179 return PSE_ALLOCATOR_DEFAULT;
180}

233void print_statistics()
234{
235 size_t total_free_memory = 0;
236 size_t largest_block = 0;
237 size_t number_of_allocations = 0;
238 size_t peak_usage = 0;
239 std::ostream& os = log::out(log::kDebug) << "statistics" << std::endl;
240 for (int i = 0; i < g_MaxAllocators; ++i)
241 {
242 if (g_Allocators[i])
243 {
244 g_Allocators[i]->get_statistics(total_free_memory, largest_block, number_of_allocations, peak_usage);
245 os << " allocator " << g_Allocators[i]->get_name() << " (" << i << "): " << std::endl;
246 os << " total free memory: " << total_free_memory / 1024 << " kbytes" << std::endl;
247 os << " largest block: " << largest_block / 1024 << " kbytes" << std::endl;
248 os << " number of allocations: " << number_of_allocations << std::endl;
249 os << " peak usage: " << peak_usage / 1024 << " kbytes" << std::endl;
250 }
251 }

254size_t get_total_free_memory()
255{
256 size_t total = 0;
257 size_t total_free_memory = 0;
258 size_t largest_block = 0;
259 size_t number_of_allocations = 0;

263 if (g_Allocators[i])

326uint32_t joaat(const std::string& str)
327{
328 uint32_t hash = 0;
329 for (const char c : str)
330 {
331 hash += c;
332 hash += (hash << 10);
333 hash ^= (hash >> 6);
334 }
335 hash += (hash << 3);
336 hash ^= (hash >> 11);
337 hash += (hash << 15);
338 return hash;
339}
