56void* CDmaPacket::AllocBuffer(int numQwords, unsigned int memMapping)
57{
58 // an alignment of 64 bytes is strictly only necessary for uncached or uncached accl mem mappings, but
59 // it ain't a bad idea in general..
60 void* mem = (void*)((uint32_t) ::operator new(numQwords * 16, std::align_val_t(64)) | memMapping);
61 // I hate to do this, but I've wasted FAR too much time hunting down cache incoherency
62 if (memMapping == Core::MemMappings::Uncached || memMapping == Core::MemMappings::UncachedAccl) {
63 // PLIN
64 FlushCache(0);
65 }
66 return mem;

97void CDmaPacket::HexDump(uint32_t numQwords)
98{
99 if (numQwords == 0)

102 printf("dumping %ld words (%ld qwords)\n", ((uint32_t)pNext - (uint32_t)pBase) / 4, numQwords);
103
104 uint32_t i = 0;
105 for (uint32_t *nextWord = (uint32_t*)pBase; nextWord != (uint32_t*)pNext; nextWord++, i++) {

107 printf("\n0x%08lx: ", (uint32_t)nextWord);
108 printf("0x%08lx ", *nextWord);
109 if (i / 4 == numQwords)

111 }

141void CSCDmaPacket::Send(bool waitForEnd, bool flushCache)
142{
143 mCheckPktLength();
144
145 // make sure we haven't forgotten to close the last dma tag
146 mAssert(pOpenTag == NULL);
147
148 // dma_channel_send_chain does NOT flush all data that is "source chained"
149 if (flushCache)
150 FlushCache(0);
151
152 // clear any memory mappings (this won't work for sp)
153 dma_channel_wait(dmaChannelId, 1000000);
154 dma_channel_send_chain(dmaChannelId, (void*)((uint32_t)pBase & 0x0fffffff), 0, bTTE ? DMA_FLAG_TRANSFERTAG : 0, 0);
155
156 if (waitForEnd)
157 dma_channel_wait(dmaChannelId, 1000000);
158}

179CVifSCDmaPacket&
180CVifSCDmaPacket::CloseUnpack(void)
181{
182 mAssert(pOpenVifCode);
183 uint32_t vn = (pOpenVifCode->cmd & 0xc) >> 2;
184 uint32_t vl = pOpenVifCode->cmd & 0x3;
185
186 // the goal here is to find the num field of the open unpack, which is the number of
187 // qwords ACTUALLY WRITTEN to vu memory (it does not count quads that are skipped in
188 // "skipping write" mode.) But first forget about skipping/filling writes and compute the number
189 // of qwords that the data in this packet will expand to.
190 uint32_t numBytes = (uint32_t)pNext - (uint32_t)pOpenVifCode - 4;
191 uint32_t numBytesPerBlock = 4 >> vl;
192 uint32_t numBlocksPerQuad = vn + 1;
193 // make sure that the data length is a multiple of 8, 16, or 32 bits, whichever is appropriate
194 mAssert((numBytes & (numBytesPerBlock - 1)) == 0);
195 uint32_t numQuads = (numBytes / numBytesPerBlock) / numBlocksPerQuad;
196
197 // We have the number of quads our data will directly expand to, so now we need to account for
198 // skipping/filling write modes.
199
200 // skipping write is easy -- we are already done
201
202 // filling write: now we get ambiguous -- what to do when numQuads == CL in the last
203 // block? I will say that in this case the vif should still do the full wl length block, filling
204 // with internal registers. If you want different behavior call CloseUnpack with a num field you've
205 // computed yourself
206 if (uiCL < uiWL) {
207 uint32_t numWLBlocks = (numQuads / uiCL);
208 uint32_t lastBlockQuads = numQuads - numWLBlocks * uiCL;
209 if (lastBlockQuads == uiCL)
210 lastBlockQuads = uiWL;
211 numQuads = numWLBlocks * uiWL + lastBlockQuads;
212 }
213
214 return CloseUnpack(numQuads);
215}
