79pse::mesh_ptr load_obj(const pse::memory::resource_id& id, const std::string& objFile, const std::string& dir)
80{
81 pse::mesh_ptr mesh = g_RenderManager->mModelManager->create_mesh(id);
82 tinyobj::attrib_t attributes;
83 std::vector<tinyobj::shape_t> shapes;
84 std::vector<tinyobj::material_t> materials;
85 std::string warnings;
86 std::string errors;
87 tinyobj::LoadObj(&attributes, &shapes, &materials, &warnings, &errors, objFile.c_str(), dir.c_str());
88
89 std::map<std::string, std::pair<pse::material_ptr, pse::texture_ptr>> lut;
90
91 for(auto& material : materials)
92 {
93 pse::math::color ambient = pse::math::color(material.ambient[0], material.ambient[1], material.ambient[2]);
94 pse::math::color diffuse = pse::math::color(material.diffuse[0], material.diffuse[1], material.diffuse[2]);
95 pse::math::color specular = pse::math::color(material.specular[0], material.specular[1], material.specular[2]);
96 pse::math::color emmision = pse::math::color(material.emission[0], material.emission[1], material.emission[2]);
97 pse::material_ptr mat = g_RenderManager->mMaterialManager->create_material(id.mName + material.name, ambient, diffuse, specular, emmision, material.shininess, 0);
98 pse::texture_ptr tex = nullptr;
99 if(!material.diffuse_texname.empty())
100 {
101 tex = g_RenderManager->mTextureManager->load_gs_texture(dir + "/" + material.diffuse_texname, dir + "/" + material.diffuse_texname);
102 }
103 lut[material.name] = std::pair<pse::material_ptr, pse::texture_ptr>(mat, tex);
104 }
105
106 for(auto& shape : shapes)
107 {
108 if(shape.mesh.material_ids.empty()) continue;
109 std::pair<pse::material_ptr, pse::texture_ptr> mat = lut[materials[shape.mesh.material_ids[0]].name];
110 pse::sub_mesh& sub = mesh->mSubMeshes.emplace_back(mat.first, mat.second, GL_TRIANGLES, shape.mesh.indices.size(), shape.mesh.indices.size());
111 int j = 0;
112 for(auto& i : shape.mesh.indices)
113 {
114 pse::math::vec3 position = {
115 attributes.vertices[i.vertex_index * 3],
116 attributes.vertices[i.vertex_index * 3 + 1],
117 attributes.vertices[i.vertex_index * 3 + 2]
118 };
119 pse::math::vec3 normal = {
120 attributes.normals[i.normal_index * 3],
121 attributes.normals[i.normal_index * 3 + 1],
122 -attributes.normals[i.normal_index * 3 + 2]
123 };
124 pse::math::vec2 texCoord = {
125 attributes.texcoords[i.texcoord_index * 2],
126 1.0F - attributes.texcoords[i.texcoord_index * 2 + 1],
127 };
128 sub.mVertexBuffer[j] = position;
129 sub.mNormalBuffer[j] = normal;
130 sub.mTexCoordBuffer[j] = texCoord;
131 j++;
132 }
133 }
134 return mesh;
135}
