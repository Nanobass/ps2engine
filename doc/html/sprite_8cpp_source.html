125void CSprite::Draw(CVifSCDmaPacket& packet)
126{
127 packet.Cnt();
128 {
129 // the data needs to be qword-aligned, so pad with appropriate # of vifnops to
130 // put the direct vifcode at the end of a qword
131 packet.Nop();

133 packet.Nop().Nop();
134 }

138 packet.CloseDirect();
