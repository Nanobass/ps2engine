968mat4 mat4::inverted() const

970

972 mat4 src = transposed();
973
974 /* calculate pairs for first 8 elements (cofactors) */
975 float tmp[12]; /* temp array for pairs */
976 tmp[0] = src.matrix[10] * src.matrix[15];
977 tmp[1] = src.matrix[11] * src.matrix[14];
978 tmp[2] = src.matrix[9] * src.matrix[15];

986 tmp[10] = src.matrix[8] * src.matrix[13];

1007
1008 /* calculate pairs for second 8 elements (cofactors) */
1009 tmp[0] = src.matrix[2] * src.matrix[7];
1010 tmp[1] = src.matrix[3] * src.matrix[6];
1011 tmp[2] = src.matrix[1] * src.matrix[7];
1012 tmp[3] = src.matrix[3] * src.matrix[5];
1013 tmp[4] = src.matrix[1] * src.matrix[6];

1016 tmp[7] = src.matrix[3] * src.matrix[4];
1017 tmp[8] = src.matrix[0] * src.matrix[6];
1018 tmp[9] = src.matrix[2] * src.matrix[4];
1019 tmp[10] = src.matrix[0] * src.matrix[5];
1020 tmp[11] = src.matrix[1] * src.matrix[4];
1021
1022 /* calculate second 8 elements (cofactors) */
1023 dst.matrix[8] = tmp[0] * src.matrix[13] + tmp[3] * src.matrix[14] + tmp[4] * src.matrix[15];
1024 dst.matrix[8] -= tmp[1] * src.matrix[13] + tmp[2] * src.matrix[14] + tmp[5] * src.matrix[15];
1025 dst.matrix[9] = tmp[1] * src.matrix[12] + tmp[6] * src.matrix[14] + tmp[9] * src.matrix[15];

1038 dst.matrix[15] -= tmp[8] * src.matrix[9] + tmp[11] * src.matrix[10] + tmp[5] * src.matrix[8];

1044 det = 1 / det;
1045 for (int j = 0; j < 16; j++)
1046 dst.matrix[j] *= det;
1047
1048 return dst;
1049}
