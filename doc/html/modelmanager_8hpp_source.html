72 void set_vertex_arrays()
73 {
74 // we will always have these
75 glEnableClientState(GL_VERTEX_ARRAY);
76 glVertexPointer(3, GL_FLOAT, 0, mVertexBuffer.data());
77 // not having normals will blow up our dma packets
78 glEnableClientState(GL_NORMAL_ARRAY);
79 if(mNormalBuffer.emtpy()) pglNormalPointer(3, GL_FLOAT, 0, mVertexBuffer.data());
80 else pglNormalPointer(3, GL_FLOAT, 0, mNormalBuffer.data());
81 // not having texcoords will also blow up our dma packets
82 glEnableClientState(GL_TEXTURE_COORD_ARRAY);
83 if(mTexCoordBuffer.emtpy())
84 {
85 glTexCoordPointer(2, GL_FLOAT, 0, mVertexBuffer.data());
86 glDisable(GL_TEXTURE_2D);
87 }
88 else
89 {
90 glTexCoordPointer(2, GL_FLOAT, 0, mTexCoordBuffer.data());
91 glEnable(GL_TEXTURE_2D);
92 }
93 // colors are not duplicated if not present, good job ps2gl
94 if(mColorBuffer.emtpy()) glDisableClientState(GL_COLOR_ARRAY);
95 else { glEnableClientState(GL_COLOR_ARRAY); glColorPointer(4, GL_FLOAT, 0, mColorBuffer.data()); }
96 }

98 void draw_arrays()
99 {
100 // apply new material and texture if needed
101 if(mMaterial) mMaterial->apply();
102 if(mTexture) mTexture->bind();
103 // setup buffers
104 set_vertex_arrays();
105 // do some drawing
106 glDrawArrays(mPrimType, 0, mPrimCount);
107 }

109 void generate_display_list()
110 {
111 // generate list
112 if(mDList) { glDeleteLists(mDList, 1); mDList = 0; }
113 if(!mDList) mDList = glGenLists(1);
114 glNewList(mDList, GL_COMPILE);
115 // apply new material and texture if needed
116 if(mMaterial) mMaterial->apply();
117 if(mTexture) mTexture->bind();
118 // do some drawing
119 glDrawArrays(mPrimType, 0, mPrimCount);
120 glEndList();
121 }
