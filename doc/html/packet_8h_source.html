230 * CDmaPacket inlines

232
233#define mCheckFreeSpace(__type) \
234 mErrorIf(pNext + sizeof(__type) > (pBase + uiBufferQwordSize * 16), "Not enough space in packet!")
235#define mCheckFreeSpaceN(__type, __num) \
236 mErrorIf(pNext + (__num) * sizeof(__type) > (pBase + uiBufferQwordSize * 16), "Not enough space in packet!")
237
238#define mCheckPktAlignment(__type) \
239 mWarnIf(sizeof(__type) == 16 \
240 && (uint32_t)pNext & (16 - 1), \
241 "You're trying to add 16-byte data to this packet on a non-16-byte boundary.. Are you sure this is right?")
242
243#define mAddData(__type, __data) \
244 *reinterpret_cast<__type*>(pNext) = __data; \
245 pNext += sizeof(__type);
246
247template <class dataType>
248inline void
249CDmaPacket::operator+=(const dataType data)
250{
251 mCheckFreeSpace(dataType);
252 mCheckPktAlignment(dataType);
253
254 mAddData(dataType, data);
255}
256
257template <class dataType>
258inline dataType*
259CDmaPacket::Add(const dataType data)
260{
261 mCheckFreeSpace(dataType);
262 mCheckPktAlignment(dataType);
263
264 dataType* dataStart = reinterpret_cast<dataType*>(pNext);
265 mAddData(dataType, data);
266 return dataStart;
267}
268
269template <class dataType>
270inline dataType*
271CDmaPacket::Add(const dataType* data, uint32_t num)
272{
273 mCheckFreeSpaceN(dataType, num);
274 mCheckPktAlignment(dataType);
275
276 dataType* dataStart = reinterpret_cast<dataType*>(pNext);
277 dataType* nextStore = dataStart;
278 for (uint32_t i = 0; i < num; i++)
279 *nextStore++ = *data++;
280 pNext = reinterpret_cast<uint8_t*>(nextStore);
281
282 return dataStart;
283}

285inline void
286CDmaPacket::operator+=(const CDmaPacket& otherPkt)
287{
288 uint32_t numBytes = otherPkt.GetByteLength();
289 mErrorIf(numBytes & (16 - 1), "Can only add packets that are an even # of quads.");
290
291 uint32_t numQuads = numBytes / 16;
292 Add(otherPkt.GetBase(), numQuads);
293}

417inline CSCDmaPacket&
418CSCDmaPacket::CloseTag(void)
419{
420 mErrorIf(!pOpenTag, "You called CloseTag(), but no dma tags are open!");
421 mErrorIf(((uint32_t)pNext & (16 - 1)) != 0, "Packet is not qword aligned");
422 // set the qwc field of any open tags.. (- 1 is so that we don't count the qword
423 // containing the *pOpenTag)
424 if (pOpenTag)
425 pOpenTag->QWC = (((uint32_t)pNext - (uint32_t)pOpenTag) / 16 - 1);
426
427 pOpenTag = NULL;
428 return *this;
429}
430
431inline void
432CSCDmaPacket::AddDmaTag(uint32_t QWC, uint32_t PCE, uint32_t ID, uint32_t IRQ, const uint128_t* ADDR, uint32_t SPR)
433{
434 mErrorIf(((uint32_t)pNext & 0xf) != 0, "Free space in packet is not aligned properly.");
435 mErrorIf(pOpenTag, "You need to close any open dma tags before opening another!");
436
437 if (QWC == countQWC) {
438 SetDmaTag((tDmaTag*)pNext, 0, PCE, ID, IRQ, ADDR, SPR);
439 pOpenTag = (tDmaTag*)pNext;
440 } else {
441 SetDmaTag((tDmaTag*)pNext, QWC, PCE, ID, IRQ, ADDR, SPR);
442 pOpenTag = NULL;
443 }
444
445 if (bTTE) {
446 pNext += 8;
447 uiTTEBytesLeft = 8;
448 } else
449 pNext += 16;
450}
