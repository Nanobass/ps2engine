60 gsrTrxPos.src_y = 0;
61 gsrTrxPos.dest_x = 0;
62 gsrTrxPos.dest_y = 0;
63 gsrTrxPos.direction = 0; // upper-left -> lower-right (only for xfers within gs mem)
64
65 // dimensions of xfer area
66 gsrTrxReg.trans_w = 0;
67 gsrTrxReg.trans_h = 0;
68
69 // setup the giftag for sending image transfer settings to the gs
70 ImageXferSettingsGifTag.NLOOP = uiNumXferGSRegs;
71 ImageXferSettingsGifTag.EOP = 0;
72 ImageXferSettingsGifTag.PRE = 0;

81void CImageUploadPkt::BuildXferTags(void)
82{
83 // see comment above this function
84 SetTTE(true);
85
86 // transfer the registers
87 this->Cnt();
88 this->Nop().OpenDirect();
89 pNext += 5 * 16;
90 this->CloseDirect().CloseTag();
91
92 // some giftag defaults
93 // PLIN
94 // sceGifTag imageDataGifTag;
95 // SCE_GIF_CLEAR_TAG( &imageDataGifTag );
96 tGifTag imageDataGifTag = { 0, 0, 0, 0, 0, 0, 0, 0, 0 };
97 imageDataGifTag.FLG = 2; // image mode
98 imageDataGifTag.EOP = 0;
99 imageDataGifTag.NLOOP = 0;
100
101 mAssert(pImage != NULL);
102 uint128_t* image = pImage;
103
104 bool imageOnSP = false;
105
106 // is the image on the scratchpad?
107 imageOnSP = ((uint32_t)pImage & 0x70000000);
108 if (imageOnSP) {
109 image = (uint128_t*)((uint32_t)image & 0x3ff0);
110 }
111
112 uint32_t width = gsrTrxReg.trans_w, height = gsrTrxReg.trans_h;
113 uint32_t bytesInImage = width * height * GS::GetBitsPerPixel((GS::tPSM)gsrBitBltBuf.dest_pixmode) / 8;
114
115 uint32_t numQuadsInImage = ((bytesInImage & 0xf) == 0) ? bytesInImage / 16 : bytesInImage / 16 + 1;
116 uint32_t numQuadsLeft = numQuadsInImage;
117 const uint32_t maxQuadsPerGT = (1 << 15) - 1; // limited by the NLOOP field
118
119 while (numQuadsLeft > 0) {
120 // set up giftag
121 uint32_t numQuadsThisGT = std::min(numQuadsLeft, maxQuadsPerGT);
122 imageDataGifTag.NLOOP = numQuadsThisGT;
123 imageDataGifTag.EOP = (numQuadsThisGT == numQuadsLeft) ? 1 : 0;
124
125 // xfer giftag
126 this->Cnt();
127 this->Nop().OpenDirect();
128 *this += imageDataGifTag;
129 this->CloseDirect();
130 this->CloseTag();
131
132 if (numQuadsThisGT == numQuadsLeft)
133 this->Refe(&image[numQuadsInImage - numQuadsLeft], numQuadsThisGT, Packet::kNoIrq, imageOnSP);
134 else
135 this->Ref(&image[numQuadsInImage - numQuadsLeft], numQuadsThisGT, Packet::kNoIrq, imageOnSP);
136
137 // these will fit in the upper 64 bits after the dma tag (ref or refe)
138 this->Nop().OpenDirect().CloseDirect(numQuadsThisGT);
139
140 numQuadsLeft -= numQuadsThisGT;
141 }
142
143 // see comment above this function
144 SetTTE(false);
145}

147void CImageUploadPkt::Send(CSCDmaPacket& packet)
148{
149 mErrorIf(packet.GetTTE(), "Only vif source chain packets can use this class to xfer images with tte on.");
150
151 uint128_t* thisPktCopy = packet.Add((CDmaPacket&)*this);
152
153 // change the last refe in the copy to a ref
154 tSourceChainTag* lastRefe = reinterpret_cast<tSourceChainTag*>(thisPktCopy + this->GetByteLength() / 16 - 1);
155 mErrorIf(lastRefe->ID != DMAC::kRefe, "Something ain't right..");
156 lastRefe->ID = DMAC::kRef;
157}
