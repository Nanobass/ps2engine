28void PadDevice::Open(int port, int slot)
29{
30 if(mState == Connected)
31 {
32 Close();
33 }
34 mPort = port;
35 mSlot = slot;
36 mState = Waiting;
37 printf("Pad[%d, %d]\n", mPort, mSlot);
38 int ret;
39 if(!(ret = padPortOpen(mPort, mSlot, mPadBuf)))
40 {
41 mState = Disconnected;
42 printf("Can't Open Pad[%d, %d] ret=%d\n", mPort, mSlot, ret);
43 }
44 WaitReady();
45
46 int modes = padInfoMode(mPort, mSlot, PAD_MODETABLE, -1);
47 if(!modes)
48 {
49 mState = Disconnected;
50 printf("Pad[%d, %d] is not a 'DualShock 2' Controller\n", mPort, mSlot);
51 }
52
53 // Verify that the controller has a DUAL SHOCK mode
54 int i = 0;
55 do {
56 if (padInfoMode(mPort, mSlot, PAD_MODETABLE, i) == PAD_TYPE_DUALSHOCK) break;
57 i++;
58 } while (i < modes);
59
60 if(i >= modes)
61 {
62 mState = Disconnected;
63 printf("Pad[%d, %d] is not a 'DualShock 2' Controller\n", mPort, mSlot);
64 }
65
66 // If ExId != 0x0 => This controller has actuator engines
67 // This check should always pass if the Dual Shock test above passed
68 ret = padInfoMode(mPort, mSlot, PAD_MODECUREXID, 0);
69 if(!ret)
70 {
71 mState = Disconnected;
72 printf("Pad[%d, %d] is not a 'DualShock 2' Controller\n", mPort, mSlot);
73 }
74
75 printf("Enabling 'DualShock 2' functions\n");
76
77 // When using MMODE_LOCK, user cant change mode with Select button
78 padSetMainMode(mPort, mSlot, PAD_MMODE_DUALSHOCK, PAD_MMODE_LOCK);
79
80 WaitReady();
81 if(padInfoPressMode(mPort, mSlot)) printf("'DualShock 2' has Pressure Sensitive Buttons\n");
82
83 WaitReady();
84 padEnterPressMode(mPort, mSlot); // Set pressure sensitive mode
85
86 WaitReady();
87 mActuators = padInfoAct(mPort, mSlot, -1, 0);
88 printf("Pad[%d, %d] has %d Actuator/s\n", mPort, mSlot, mActuators);
89 if (mActuators != 0) {
90 mActAlign[0] = 0; // Enable small engine
91 mActAlign[1] = 1; // Enable big engine
92 mActAlign[2] = 0xff;
93 mActAlign[3] = 0xff;
94 mActAlign[4] = 0xff;
95 mActAlign[5] = 0xff;
96 WaitReady();
97 printf("Pad[%d, %d] align=%d\n", mPort, mSlot, padSetActAlign(mPort, mSlot, mActAlign));
98 }
99
100 WaitReady();
101 printf("Pad[%d, %d] Initialized\n", mPort, mSlot);
102
103 mState = Connected;
104}

113void PadDevice::Poll()
114{
115 int ret = padGetState(mPort, mSlot);
116 if (ret == PAD_STATE_DISCONN)
117 {
118 printf("Pad[%d, %d] disconnected\n", mPort, mSlot);
119 mState = Disconnected;
120 }
121
122 if((ret != PAD_STATE_STABLE) && (ret != PAD_STATE_FINDCTP1)) return;
123
124 ret = padRead(mPort, mSlot, &mPadButtons);
125 if(ret)
126 {
127 mPadData = 0xFFFF ^ mPadButtons.btns;
128 mOldPad = mNewPad;
129 mNewPad = mPadData;
130
131 mInputs[Cross] = Input(mPadButtons.cross_p, 0, 255, 0, 16);
132 mInputs[Circle] = Input(mPadButtons.circle_p, 0, 255, 0, 16);
133 mInputs[Square] = Input(mPadButtons.square_p, 0, 255, 0, 16);
134 mInputs[Triangle] = Input(mPadButtons.triangle_p, 0, 255, 0, 16);
135
136 mInputs[Start] = Input(mPadData & PAD_START ? 255 : 0, 0, 255, 0, 16);
137 mInputs[Select] = Input(mPadData & PAD_SELECT ? 255 : 0, 0, 255, 0, 16);
138
139 mInputs[Left] = Input(mPadButtons.left_p, 0, 255, 0, 16);
140 mInputs[Right] = Input(mPadButtons.right_p, 0, 255, 0, 16);
141 mInputs[Up] = Input(mPadButtons.up_p, 0, 255, 0, 16);
142 mInputs[Down] = Input(mPadButtons.down_p, 0, 255, 0, 16);
143
144 mInputs[L1] = Input(mPadButtons.l1_p, 0, 255, 0, 16);
145 mInputs[L2] = Input(mPadButtons.l2_p, 0, 255, 0, 16);
146 mInputs[R1] = Input(mPadButtons.r1_p, 0, 255, 0, 16);
147 mInputs[R2] = Input(mPadButtons.r2_p, 0, 255, 0, 16);
148
149 mInputs[L3] = Input(mPadData & PAD_L3 ? 255 : 0, 0, 255, 0, 16);
150 mInputs[R3] = Input(mPadData & PAD_R3 ? 255 : 0, 0, 255, 0, 16);
151
152 mInputs[DpadX] = Input(-mPadButtons.left_p + mPadButtons.right_p, 0, 255, 0, 16);
153 mInputs[DpadY] = Input(-mPadButtons.down_p + mPadButtons.up_p, 0, 255, 0, 16);
154
155 mInputs[LeftX] = Input(mPadButtons.ljoy_h, 127, 128, 16, 32);
156 mInputs[LeftY] = Input(mPadButtons.ljoy_v, 127, 128, 16, 32);
157 mInputs[RightX] = Input(mPadButtons.rjoy_h, 127, 128, 16, 32);
158 mInputs[RightY] = Input(mPadButtons.rjoy_v, 127, 128, 16, 32);
159
160 }

182void PadDevice::SetRumble(uint32_t rumble, float value)
183{
184 switch(rumble)
185 {
186 case Small: mActAlign[Small] = (int) (value * 255); break;
187 case Big: mActAlign[Big] = (int) (value * 255); break;
188 default: break;
189 }
190 padSetActDirect(mPort, mSlot, mActAlign);

193void PadDevice::WaitReady()
194{
195 ConnectionState old = mState;
196 mState = Waiting;
197 int state, lastState;
198 char stateString[16];
199 state = padGetState(mPort, mSlot);
200 lastState = -1;
201 while ((state != PAD_STATE_STABLE) && (state != PAD_STATE_FINDCTP1)) {
202 if (state != lastState) {
203 padStateInt2String(state, stateString);
204 printf("Pad[%d, %d] state changed: %s\n", mPort, mSlot, stateString);
205 }
206 lastState = state;
207 state = padGetState(mPort, mSlot);
208 }
209 if (lastState != -1) printf("Pad[%d, %d] is Ready\n", mPort, mSlot);
210 mState = old;
211}
